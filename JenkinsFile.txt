pipeline {
    agent any
   
    stages {
        stage('Starting') {
            steps {
                echo 'Started pipeline'
            }
        }
        stage('git'){
            steps{
               git branch: 'main', credentialsId: 'github-pwd', url: 'gitURL'
            }
        }
        stage('clean'){
            steps{
                bat 'mvn clean'
            }
        }
        stage('Test'){
            steps{
                bat 'mvn test'
            }
        }
    }
}
